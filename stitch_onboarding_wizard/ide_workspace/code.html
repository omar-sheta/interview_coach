def bubble_sort(arr):
    # Simple implementation of bubble sort
    n = len(arr)
    
    for i in range(n):
        # Track if swap happened
        swapped = False
        
        for j in range(0, n - i - 1):
            if arr[j] > arr[j + 1]:
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
                swapped = True
        
        if not swapped:
            break
            
    return arr
